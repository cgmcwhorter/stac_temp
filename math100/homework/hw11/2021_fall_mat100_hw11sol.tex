% Problem 1
\problem{10} Solve $x^2 + 2x - 3= 0$ by factoring. Show all your work. \pspace

\sol
	\[
	\begin{aligned}
	x^2 + 2x - 3&= 0 \\[0.3cm]
	(x + 3)(x - 1)&= 0
	\end{aligned}
	\] \pspace
But then either $x + 3= 0$, which implies $x= -3$, or $x - 1=0$, which implies $x= 1$. Therefore, $x= -3, 1$.

\sol
	\[
	\begin{aligned}
	x^2&= 6x - 9 \\[0.3cm]
	x^2 - 6x + 9&= 0 \\[0.3cm]
	(x - 3)(x - 3)&= 0 \\[0.3cm]
	(x - 3)^2&= 0 \\[0.3cm]
	\sqrt{(x - 3)^2}&= \sqrt{0} \\[0.3cm]
	x - 3&= 0 \\[0.3cm]
	x&= 3
	\end{aligned}
	\]

% Problem 6
\problem{10} Solve $x^2 - 7x - 18= 0$ using the quadratic formula. Show all your work. \pspace

\sol
	\[
	\begin{aligned}
	x&= \dfrac{-b \pm \sqrt{b^2 - 4ac}}{2a} \\[0.3cm]
	x&= \dfrac{-(-7) \pm \sqrt{(-7)^2 - 4(1)(-18)}}{2(1)} \\[0.3cm]
	x&= \dfrac{7 \pm \sqrt{49 + 72}}{2} \\[0.3cm]
	x&= \dfrac{7 \pm \sqrt{121}}{2} \\[0.3cm]
	x&= \dfrac{7 \pm 11}{2} \\[0.3cm]
	x&= \dfrac{7 + 11}{2}, \dfrac{7 - 11}{2} \\[0.3cm]
	x&= \dfrac{18}{2}, \dfrac{-4}{2} \\[0.3cm]
	x&= -2, 9
	\end{aligned}
	\]
