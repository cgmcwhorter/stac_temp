	f(x)&= 1 - 4x \\[0.3cm]

	\end{aligned}
	\]

        \begin{enumerate}[(a)]
        \item $6f(1) - g(2)$ \vfill
        \item $(f + g)(1)$ \vfill
        \item $(f - g)(0)$ \vfill
        \item $(fg)(2)$ \vfill
        \item $(f \circ g)(-1)$ \vfill
        \item $(g \circ f)(-1)$ \vfill
        \end{enumerate}
