	\[

	\hspace{-0.7cm} \max z= 4x_1 - x_2 + 5x_3 \\

	-x_1 + x_2 - 5x_3 \geq 30 \\
	x_1, x_2, x_3 \geq 0
	\end{cases}
	\end{gathered}
	\]
