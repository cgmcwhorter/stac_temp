% Problem 1
\problem{10} Define the following propositions:
	\begin{table}[H]
	\centering
	\begin{tabular}{rl}
	$P$: & Bill took a Math course. \\
	$Q$: & Susan is not a Biology major. \\
	$R$: & Bill is a senior. \\
	$S$: & Susan is a Sophomore. 
	\end{tabular}
	\end{table}
Write the following logical propositions as a complete English sentence:
	\begin{enumerate}[(a)]
	\item $P \wedge Q$
	\item $P \vee R$
	\item $S \wedge \neg Q$
	\item $R \to P$
	\end{enumerate}

	\item $P \to Q$

	\end{enumerate}
