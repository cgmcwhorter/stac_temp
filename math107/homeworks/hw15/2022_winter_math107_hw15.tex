% Problem 1
\problem{10} Let $A$ and $B$ be events in a probability space with\dots
	\[
	\begin{gathered}
	P(A)= 0.45 \\
	P(B)= 0.23 \\
	P(A \text{ and } B)= 0.16
	\end{gathered}
	\]

\begin{enumerate}[(a)]
\item Find $P(A \text{ or } B)$.
\item Are $A$ and $B$ disjoint? Explain. 
\item Are $A$ and $B$ independent? Explain. 
\end{enumerate}
