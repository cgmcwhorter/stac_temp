\problem{10} Suppose you have a count, $X$, from the binomial distribution $B(9, 0.15)$. Find the following:
	\begin{enumerate}[(a)]
	\item $P(X= 3)$
	\item $P(X < 2)$
	\item $P(X \geq 4)$
	\item $P(1 \leq X < 5)$
	\end{enumerate}
