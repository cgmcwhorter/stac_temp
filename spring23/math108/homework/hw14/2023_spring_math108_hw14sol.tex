% Problem 1
\problem{10} Consider the following system of equations:
	\[
	\begin{aligned}
	3x - 2y&= -8 \\
	-x + 3y&= 5
	\end{aligned}
	\]

\begin{enumerate}[(a)]
\item Find the coefficient matrix, $A$. 
\item Show that $A$ has an inverse. 
\item Use your answer from (b) to find the solution to the system of equations. 
\end{enumerate}
