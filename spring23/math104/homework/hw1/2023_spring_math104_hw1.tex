% Problem 1
\problem{10} Define a function $f(x)$ by $f(x)= x^2 + 4x - 2$. Showing all your work and simplifying as much as possible, find the following:
	\begin{enumerate}[(a)]
	\item $f(0)$
	\item $f(-2)$
	\item $f(x - 2)$
	\item $f(x + h)$
	\item $\dfrac{f(x + h) - f(x)}{h}$
	\end{enumerate}

% Problem 2
\problem{10} Let $f(x)= 2x - 5$ and $g(x)= x^2 - 6$. Showing all your work, compute the following:
	\begin{enumerate}[(a)]
	\item $(f - g)(-2)$
	\item $(fg)(1)$
	\item $\left( \dfrac{f}{g} \right)(3)$
	\item $(f \circ g)(0)$
	\item $(g \circ f)(0)$
	\end{enumerate}
