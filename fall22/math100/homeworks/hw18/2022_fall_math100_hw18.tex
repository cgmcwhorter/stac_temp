\problem{10} Consider the normal distribution $N(125, 18)$. Find the following:
	\begin{enumerate}[(a)]
	\item $P(X= 125)$
	\item $P(X \leq 125)$
	\item $P(X \leq 150)$
	\item $P(X \geq 150)$
	\item $P(125 \leq X \leq 150)$
	\end{enumerate}
