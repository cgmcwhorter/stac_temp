	\[
	\begin{aligned}
	\text{max } z= 4x_1& + 6x_2 \\
	-x_1 + 5x_2&\leq 5 \\
	x_1 + x_2&\leq 11 \\
	x_1, x_2&\geq 0 \\
	x_2&\leq 7
	\end{aligned}
	\]
	\end{minipage}\begin{minipage}{0.69\textwidth}
	\[
	\fbox{
	\begin{tikzpicture}[scale=1.5,every node/.style={scale=0.5}]

	ytick={0,2,4,6,8,10},
	minor tick = {-1,0,1,...,10},

	\end{axis}
