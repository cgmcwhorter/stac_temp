% Problem 1
\problem{10} Factor $x^2 + 7x - 30$. Show all your work. \pspace

\sol 
	\begin{table}[!ht]
	\centering
	\underline{\bfseries 30} \pvspace{0.2cm}
	\begin{tabular}{rr}
	$1 \cdot -30$ & $-29$ \\
	$-1 \cdot 30$ & $29$ \\
	$2 \cdot -15$ & $-13$ \\
	$-2 \cdot 15$ & $13$ \\
	$3 \cdot -10$ & $-7$ \\ \hline
	\multicolumn{1}{|r}{$-3 \cdot 10$} & \multicolumn{1}{r|}{$7$} \\ \hline
	$5 \cdot -6$ & $-1$ \\
	$-5 \cdot 6$ & $1$ 
	\end{tabular}
	\end{table}

Therefore,
	\[
	x^2 + 7x - 30= (x - 3)(x + 10)
	\]
