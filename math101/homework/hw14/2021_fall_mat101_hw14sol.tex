% Problem 1
\problem{10} Compute the following, being sure to show all your work and simplifying as much as possible:
	\[
	\dfrac{5}{x - 1} - \dfrac{4}{x + 3}
	\] \pspace

\sol
	\[
	\begin{aligned}
	\dfrac{5}{x - 1} - \dfrac{4}{x + 3}&= \dfrac{5(x + 3)}{(x - 1)(x + 3)} - \dfrac{4(x - 1)}{(x - 1)(x + 3)} \\[0.3cm]
	&= \dfrac{5(x + 3) - 4(x - 1)}{(x - 1)(x + 3)} \\[0.3cm]
	&= \dfrac{5x + 15 - 4x + 4}{(x - 1)(x + 3)} \\[0.3cm]
	&= \dfrac{x + 19}{(x - 1)(x + 3)}
	\end{aligned}
	\]
